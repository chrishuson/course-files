\subsubsection*{6.7 Quiz: Non-right triangle trigonometry \hfill HSG.SRT.D.11}
\emph{Round all values to three significant figures.}
\begin{enumerate}
\item Right triangle $ABC$ is shown with $AB=15$, $m\angle A=37^\circ$. Find the value of $BC=x$.
  \begin{flushright}
    \begin{tikzpicture}[scale=0.6]
      \draw [thick](0,0)node[below]{$A$}--
      (6,0)node[below]{$C$}--
      (6,4.5)node[above right]{$B$}--cycle;
      \draw (6,0)++(-0.6,0)--++(0,0.6)--+(0.6,0);
      \node at (3,3){$15$};
      \node at (6,2.5)[right]{$x$};
      \draw [thick, -] (1,0) arc [start angle=0, end angle=37, radius=1];
      \node at (1.8,0)[above]{$37^\circ$};
    \end{tikzpicture}
  \end{flushright}

\item Given $\triangle ABC$ with $AC=28$ centimeters, base $AB=40$, and $\hat{A} = 62^\circ$.
\begin{multicols}{2}
  \begin{enumerate}
    \item Find altitude $h$ cm using $\displaystyle \sin \hat{A}=\frac{h}{28}$.
    \item Find the area of the triangle\\[0.5cm]
    $Area = \frac{1}{2}bh$
  \end{enumerate}
  \begin{tikzpicture}[scale=1]
    \draw [thick]
      (0,0)node[below]{$A$}--
      (6,0)node[below]{$B$}--
      (55:4)node[above]{$C$} --cycle;
    \draw [dashed] (55:4)--(2.3,0);
    \draw (2.3,0)++(0.3,0)--++(0,0.3)--+(-0.3,0);
    \node at (2.5,1.2)[right]{$h$};
    \node at (1,2){$28$};
    \node at (3,0)[below]{$40$};
    \draw [thick, -] (1,0) arc [start angle=0, end angle=55, radius=1];
    \node at (0.6,0)[above]{$62^\circ$};
    \node at (38:6.5){\emph{diagram not to scale}};
  \end{tikzpicture}  
\end{multicols}
\vspace{2.5cm}
