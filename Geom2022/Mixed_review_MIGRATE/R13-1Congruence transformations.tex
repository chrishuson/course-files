\begin{multicols}{2}
\begin{enumerate}
    \item $y-1=\frac{1}{2}(x+1)$
    \item $y-1=-\frac{1}{2}(x+1)$ 
    \item $y+1=\frac{1}{2}(x-1)$
    \item $y+1=-\frac{1}{2}(x-1)$
\end{enumerate}
\end{multicols}
