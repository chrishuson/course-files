\item The expression $(x + a)(x + b)$ can not be written as
\begin{enumerate}
    \item $a(x + b)+ x(x + b)$
    \item $x^2 + (a + b)x + ab$ 
    \item  $x^2 + abx + ab$  
    \item $x(x + a)+ b(x + a)$
\end{enumerate}
