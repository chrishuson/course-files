  1 meter = 39.37 inches\\
  1 mile = 5280 feet\\
  1 mile = 1760 yards\\
  1 mile = 1.609 kilometers\\
  1 kilometer = 0.62 mile\\
  1 pound = 16 ounces\\
  1 pound = 0.454 kilogram \\
  1 kilogram = 2.2 pounds \\
  1 ton = 2000 pounds\\
  1 cup = 8 fluid ounces \\
  1 pint = 2 cups\\
  1 quart = 2 pints\\
  1 gallon = 4 quarts\\
  1 gallon = 3.785 liters\\
  1 liter = 0.264 gallon\\
  1 liter = 1000 cubic centimeters
\end{multicols}
