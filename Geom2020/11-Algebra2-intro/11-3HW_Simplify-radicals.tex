\subsubsection*{11.3 Homework: Simplify radicals \\
\emph{Do not use a calculator or convert values to decimals}}

\begin{enumerate}

\subsubsection*{Simplify each expression}
\item 
\begin{enumerate}[itemsep=3cm]
  \begin{multicols}{2}
    \item $\sqrt{45}$
    \item $\sqrt{\theta^2} - 2 \beta + 7\theta$, $\theta>0$
    \item $\sqrt{75}+2\sqrt{3}$
    \item $2x\sqrt{7}+\sqrt{7x^2}$, $x>0$
  \end{multicols}
  \end{enumerate} \vspace{2cm}

\subsubsection*{Solve for the unknown of interest}

\item Solve for $y$ 
\begin{enumerate}
  \begin{multicols}{2}
    \item $x \sin \theta +y \cos \theta =1$
    \item $\displaystyle \frac{1}{k}x+\frac{1}{m}y=\frac{1}{n}$
  \end{multicols}
  \end{enumerate} \vspace{4cm}

\item Solve for $\theta$
\begin{enumerate}
  \begin{multicols}{2}
    \item $\theta \sin x + \theta \cos x = 1$
    \item $\theta^2+\alpha^2=\beta^2$
  \end{multicols}
  \end{enumerate}

\end{enumerate}
