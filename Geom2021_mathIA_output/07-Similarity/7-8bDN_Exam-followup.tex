      \item T \quad F \quad $\overline{AC} \cong \overline{DF}$
      \item T \quad F \quad $\displaystyle k=\frac{DF}{AC}$
      \item T \quad F \quad $\overline{AC} \rightarrow \overline{DF}$
      \item T \quad F \quad $\overline{AC} \cong \overline{DF}$
      \item T \quad F \quad $\displaystyle \frac{DE}{AB} = \frac{EF}{BC}$
    \end{enumerate}
