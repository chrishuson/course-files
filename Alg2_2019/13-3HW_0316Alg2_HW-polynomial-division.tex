\subsubsection*{\\* Homework: Polynomial long division}

\begin{enumerate}

\vspace{0.5 cm}


\item What is the quotient when $x^2-3x-40$ is divided by $x + 5$?\\*[5pt]
\item What is the quotient when $x^3+3x^2-x+2$ is divided by $x - 1$?\\*[5pt]

\item Determine whether the binomial $x+2$ is a factor of $f(x)=x^3+x^2-16x-16$\\*[5pt]

\item Use long division to determine the quotient and remainder of $(x^3+4x^2-8x-6) \div (x+2)$.\\*[5pt]

\item The expression $(x + a)(x + b)$ can not be written as
\begin{enumerate}
    \item $a(x + b)+ x(x + b)$
    \item $x^2 + (a + b)x + ab$
    \item  $x^2 + abx + ab$
    \item $x(x + a)+ b(x + a)$\\*[15pt]
\end{enumerate}



\item What is the equation of the line with slope 2 passing through the point $(1, 2)$?\\*[5pt]


\item Given the function $f(x)=(x-1)(x+3)$. State the $x$-intercepts of the graph of $f$. Find the coordinates of the vertex of the graph of $f$.\\*[5pt]




\item Given events $A$ and $B$, such that $\mathrm \mathrm P(A)=0.6, \mathrm P(B)=0.5$, and $\mathrm P(A \cup B) =0.8$, determine whether $A$ and $B$ are dependent or independent.


\end{enumerate}
