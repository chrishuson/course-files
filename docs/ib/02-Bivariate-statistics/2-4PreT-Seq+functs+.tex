\item A sequence is defined by $u_1 = 4$ and $u_{n+1} = 3u_n - 2$ for $n \ge 1$.
\begin{enumerate}[itemsep=0.1cm]
  \item Write down $u_2$ and $u_3$.
  \item Show that $u_n = 3^{n} + 1$ satisfies the recurrence.
  \item Hence find $u_6$.
\end{enumerate}
\vspace{0.1cm}

\item A geometric sequence has first term $12$ and common ratio $\frac{5}{6}$.
\begin{enumerate}[itemsep=0.1cm]
  \item Write down the second and third terms.
  \item Find the sum of the first $n$ terms, $S_n$.
  \item Find the least value of $n$ such that $S_n > 50$.
\end{enumerate}
\vspace{0.1cm}

\item A quantity $P$ decreases according to $P(t)=1800 e^{-kt}$.
\begin{enumerate}[itemsep=0.1cm]
  \item Given that $P(2)=1500$, find $k$.
  \item Find the time when $P(t)=900$.
  \item State whether the graph of $P$ is linear, quadratic or exponential.
\end{enumerate}
\vspace{0.1cm}

\item Solve the following, giving exact values when possible.
\begin{enumerate}[itemsep=0.1cm]
  \item $3^{x}=27$.
  \item $\log_{5}(2x-1)=2$.
  \item $\ln(4) - \ln(x) = \ln(2)$.
\end{enumerate}
\vspace{0.1cm}

\item Consider the function $f(x) = 2^{x} - 3$.
\begin{enumerate}[itemsep=0.1cm]
  \item Find $f(0)$ and $f(2)$.
  \item Solve $2^{x} - 3 = 5$.
  \item Describe the transformation that maps $y=2^{x}$ to $y=f(x)$.
\end{enumerate}
\vspace{0.1cm}

\item An arithmetic sequence has first term $a_1=950$ and common difference $d=25$.
\begin{enumerate}[itemsep=0.1cm]
  \item Write an expression for $a_n$.
  \item Find the smallest $n$ such that $a_n \ge 1400$.
  \item Find the sum of the first $n$ terms when $n$ is the value from part (b).
\end{enumerate}
\vspace{0.1cm}

\item A quadratic function is given by $g(x)=x^2 - 6x + 5$.
\begin{enumerate}[itemsep=0.1cm]
  \item Find the coordinates of the vertex.
  \item Solve $g(x)=0$.
  \item The line $y=mx$ intersects the graph of $g$ at two distinct points. Find the range of $m$ for which this happens.
\end{enumerate}
\vspace{0.1cm}
\end{enumerate}

\section*{Paper 2 style (GDC allowed)}

\begin{enumerate}[itemsep=0.1cm]

\item A music teacher records the number of hours $h$ students practise each week and the mark $M$ each student receives on a test.
\begin{enumerate}[itemsep=0.1cm]
  \item Use your GDC to find the product–moment correlation coefficient $r$.
  \item The regression line of $M$ on $h$ is $M = ah + b$. Write down $a$ and $b$ from your GDC output.
  \item A student currently practises $10$ hours per week. Use the regression line to estimate how many marks the student might get if they increase practice to $14$ hours per week.
\end{enumerate}
\vspace{0.1cm}

\item A school enrolment in year $1$ is $1200$ students and increases each year by $3.5\%$.
\begin{enumerate}[itemsep=0.1cm]
  \item Show that the number of students in year $n$ can be modelled by $E_n = 1200(1.035)^{n-1}$.
  \item Find the number of students in year $8$.
  \item A scholarship fund pays $75$ euros to each enrolled student each year. Find the total amount paid in the first $6$ years.
\end{enumerate}
\vspace{0.1cm}

\item The number of seats available at the school in year $n$ is given by $S_n = 1100 + 40(n-1)$.
\begin{enumerate}[itemsep=0.1cm]
  \item Write down $S_1$ and $S_{10}$.
  \item Find the least value of $n$ such that $S_n \ge E_n$ from question 9.
  \item Explain, with reference to the models, whether the school will always have enough seats for all applicants for all future years.
\end{enumerate}
\vspace{0.1cm}

\item The sum of the first $n$ terms of a geometric sequence is $S_n = 9(1 - (4/5)^n)$.
\begin{enumerate}[itemsep=0.1cm]
  \item Find the first term and the common ratio.
  \item Find $S_\infty$.
  \item Find the least value of $n$ such that $S_\infty - S_n < 0.0008$.
\end{enumerate}
\vspace{0.1cm}

\item An investment of €5000 grows at a nominal annual rate of $2.4\%$ compounded monthly.
\begin{enumerate}[itemsep=0.1cm]
  \item Write a formula for the value $V$ after $t$ years.
  \item Use your GDC to find the value after $5$ years, correct to the nearest euro.
  \item Determine the time needed for the investment to reach €6000.
\end{enumerate}
\vspace{0.1cm}

\item Consider the function $f(x)=\ln(x+2) - 1/2$ for $x \ge -1$.
\begin{enumerate}[itemsep=0.1cm]
  \item Find $f(-1)$ and $f(0)$.
  \item Solve $\ln(x+2) - 1/2 = 0$.
  \item Sketch the graph of $y=f(x)$ on $-1 \le x \le 4$.
\end{enumerate}
\vspace{0.1cm}

\item A researcher records the age of used cars (years) and their selling price (in thousands of euros).
\begin{enumerate}[itemsep=0.1cm]
  \item Find the correlation coefficient.
  \item Write down the regression equation of price on age.
  \item Comment on whether it is sensible to use this regression equation to predict the price of a brand new car.
\end{enumerate}
\vspace{0.1cm}

\item Define $f(x)=2x+3$ and $g(x)=5e^{0.2x}$.
\begin{enumerate}[itemsep=0.1cm]
  \item Find $(f\circ g)(x)$.
  \item Find $(g\circ f)(x)$.
  \item Solve $(f\circ g)(x)=33$ using your GDC.
\end{enumerate}
\vspace{0.1cm}
\end{enumerate}

\section*{Markscheme (outline)}

\textbf{1.} (a) $u_2=3\cdot 4 - 2=10$, $u_3=3\cdot 10 - 2=28$. (b) Substitute: $3(3^{n}+1)-2=3^{n+1}+1$. (c) $u_6=3^{6}+1=730$.

\textbf{2.} (a) $12$, $10$, etc. (b) $S_n = 72(1-(5/6)^n)$. (c) Solve $72(1-(5/6)^n)>50$.

\textbf{3.} (a) $1500=1800e^{-2k}\Rightarrow e^{-2k}=5/6\Rightarrow k=\tfrac{1}{2}\ln(6/5)$. (b) $900=1800e^{-kt}\Rightarrow e^{-kt}=1/2\Rightarrow t=\ln 2/k$. (c) Exponential.

\textbf{4.} (a) $x=3$. (b) $2x-1=25\Rightarrow x=13$. (c) $\ln 4 - \ln x = \ln 2 \Rightarrow x=2$.

\textbf{5.} (a) $f(0)=-3$, $f(2)=1$. (b) $2^{x}=8\Rightarrow x=3$. (c) Translation down $3$ units.

\textbf{6.} (a) $a_n=950+25(n-1)$. (b) $950+25(n-1)\ge 1400 \Rightarrow n=19$. (c) $S_{19}=\tfrac{19}{2}(1900+450)=\tfrac{19}{2}\times2350$.

\textbf{7.} (a) Vertex $(3,-4)$. (b) Roots $x=1,5$. (c) Discriminant $>0$ gives $m^2<4(1)(5)$.

\textbf{8.} Use GDC: $r$ positive, $a,b$ from regression output, substitution gives estimate.

\textbf{9.} (a) Shown algebraically. (b) $E_8=1200(1.035)^7$. (c) $6\times75\times E_{avg}$.

\textbf{10.} (a) $S_1=1100$, $S_{10}=1460$. (b) Solve $1100+40(n-1)\ge1200(1.035)^{n-1}$. (c) Geometric growth eventually exceeds linear.

\textbf{11.} (a) $u_1=9/5$, $r=4/5$. (b) $S_\infty=45$. (c) $9(4/5)^n<0.0008$.

\textbf{12.} (a) $V=5000(1+0.024/12)^{12t}$. (b) Evaluate with GDC. (c) Solve for $t$.

\textbf{13.} (a) $f(-1)=-1/2$, $f(0)=\ln 2-1/2$. (b) $x=e^{1/2}-2$. (c) Standard log curve.

\textbf{14.} Negative correlation, regression not suitable beyond data range.

\textbf{15.} (a) $(f\circ g)(x)=10e^{0.2x}+3$. (b) $(g\circ f)(x)=5e^{0.2(2x+3)}$. (c) Solve $10e^{0.2x}+3=33$.
