        x &= \frac{-5 \pm \sqrt{25 - 16}}{4} \\
        x &= \frac{-5 \pm \sqrt{9}}{4} \\
        x &= \frac{-5 \pm 3}{4} \\
        x &= \frac{-2}{4} \quad \text{or} \quad x = \frac{-8}{4} \\
