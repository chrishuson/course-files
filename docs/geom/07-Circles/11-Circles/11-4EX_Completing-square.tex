\subsubsection*{11.4 Extension: Completing the square}
  \begin{enumerate}
\item Expand each binomial-squared expression to the form $ax^2+bx+c$.
  \begin{multicols}{2}
  \begin{enumerate}[itemsep=3cm]
    \item $(x+3)(x+3)$
    \item $(x+2)^2$ 
    \item $(x+5)^2$ 
    \item $(x+7)^2$ 
  \end{enumerate}
  \end{multicols}\vspace{3cm}
  
\item Simplify each radical.
\begin{multicols}{2}
  \begin{enumerate}[itemsep=2cm]
    \item $\sqrt{50}$ 
    \item $\sqrt{18}$
    \item $\sqrt{27}$ 
    \item $\sqrt{24}$ 
  \end{enumerate}
  \end{multicols}\vspace{2cm}

\item Solve for the appropriate variable ($h$ and $r$).
  \begin{multicols}{2}
  \begin{enumerate}[itemsep=2cm]
    \item $Area=\frac{1}{2}(14.8)h=62.9$ 
    \item $Area=\pi r^2=483$ 
  \end{enumerate}
  \end{multicols}\vspace{2cm}

\newpage
\item Write down the center and radius of each circle.
  \begin{enumerate}
    \begin{multicols}{2}
    \item   $(x-4)^2+(y-3)^2=9$ \vspace{2.5cm}
    \item   $(x+5)^2+(y-2)^2=4^2$
    \item   $x^2+y^2=4$ \vspace{2.5cm}
    \item   $(x+7)^2+(y-2)^2=9^2$
    \end{multicols}
  \end{enumerate}

\end{enumerate}
