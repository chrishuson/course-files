\subsubsection*{One-Dimensional Motion and Vectors}

\item A student walks along a straight hallway (forward is $+$). Start $x_0 = 2.0$ m.
  \begin{enumerate}[itemsep=0.75cm]
    \item To $x = 7.5$ m, find displacement $\Delta x$.
    \item From $x=7.5$ m back to $x=4.0$ m, find $\Delta x$.
    \item Total displacement from start to finish.
  \end{enumerate}
  \vspace{0.5cm}

\item A cart moves from $x_1 = 1.2$ m at $t_1=0.0$ s to $x_2 = 5.8$ m at $t_2 = 3.0$ s.
  \begin{enumerate}[itemsep=0.75cm]
    \item Find displacement.
    \item Find average velocity $v_\text{avg}$.
    \item State the units for $v_\text{avg}$.
  \end{enumerate}
  \vspace{0.5cm}

\item A cyclist rides east, then west:
  \begin{enumerate}[itemsep=0.5cm]
    \item From $x=0$ m to $x=30$ m in 12 s.
    \item Then back to $x=18$ m at $t=20$ s.
    \item Find each displacement, total displacement, and total distance traveled.
  \end{enumerate}
  \vspace{0.5cm}

\item One-dimensional vector additions (east is $+$).
  \begin{multicols}{2}
  \begin{enumerate}[itemsep=0.5cm]
    \item $\Delta x_1 = +12$ m, $\Delta x_2 = -5$ m
    \item $+0.85$ km, $+1.40$ km, $-0.50$ km
  \end{enumerate}
  \end{multicols}
  \vspace{0.5cm}
  Why is distance not always equal to displacement?
  \vspace{0.5cm}

\item A car travels at $88~\text{km/h}$. The driver looks away for $1.6~\text{s}$.
  \begin{enumerate}[itemsep=1cm]
    \item Convert $88~\text{km/h}$ to m/s.
    \item How far does the car move in $1.6~\text{s}$?
  \end{enumerate}
  \vspace{0.5cm}

\item A cart starts from rest and accelerates at $0.80~\text{m/s}^2$.
  \begin{enumerate}[itemsep=1cm]
    \item Find its speed after $5.0~\text{s}$.
    \item Interpret physically what $0.80~\text{m/s}^2$ means.
    \item If displacement is $10.0$ m, find average velocity.
  \end{enumerate}
  \vspace{0.5cm}

\item A drone flies north at $12.0~\text{m/s}$ for $45.0~\text{s}$.
  \begin{enumerate}[itemsep=1cm]
    \item Convert $12.0~\text{m/s}$ to km/h.
    \item Find distance traveled.
    \item If south is negative, what sign would you assign to this displacement?
  \end{enumerate}
  \vspace{0.5cm}

\end{enumerate}
