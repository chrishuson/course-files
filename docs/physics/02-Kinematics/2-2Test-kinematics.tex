\subsubsection*{One-Dimensional Motion and Vectors}

\item A student walks along a straight hallway (forward is $+$). Start $x_0 = 1.5$ m.
  \begin{enumerate}[itemsep=0.75cm]
    \item To $x = 9.0$ m, find displacement $\Delta x$.
    \item From $x=9.0$ m back to $x=3.5$ m, find $\Delta x$.
    \item Total displacement from start to finish.
  \end{enumerate}
  \vspace{0.5cm}

\item A cart moves from $x_1 = 0.8$ m at $t_1=0.0$ s to $x_2 = 6.0$ m at $t_2 = 2.5$ s.
  \begin{enumerate}[itemsep=0.75cm]
    \item Find displacement.
    \item Find average velocity $v_\text{avg}$.
    \item State the units for $v_\text{avg}$.
  \end{enumerate}
  \vspace{0.5cm}

\item A cyclist rides east, then west:
  \begin{enumerate}[itemsep=0.5cm]
    \item From $x=0$ m to $x=36$ m in 10 s.
    \item Then back to $x=20$ m at $t=18$ s.
    \item Find each displacement, total displacement, and total distance traveled.
  \end{enumerate}
  \vspace{0.5cm}

\item One-dimensional vector additions (east is $+$).
  \begin{multicols}{2}
  \begin{enumerate}[itemsep=0.5cm]
    \item $\Delta x_1 = +15$ m, $\Delta x_2 = -8$ m
    \item $+1.10$ km, $+0.75$ km, $-0.40$ km
  \end{enumerate}
  \end{multicols}
  \vspace{0.5cm}
  Why is distance not always equal to displacement?
  \vspace{0.5cm}

\item A car travels at $96~\text{km/h}$. The driver looks away for $2.0~\text{s}$.
  \begin{enumerate}[itemsep=1cm]
    \item Convert $96~\text{km/h}$ to m/s.
    \item How far does the car move in $2.0~\text{s}$?
  \end{enumerate}
  \vspace{0.5cm}

\item A cart starts from rest and accelerates at $1.2~\text{m/s}^2$.
  \begin{enumerate}[itemsep=1cm]
    \item Find its speed after $4.0~\text{s}$.
    \item Interpret physically what $1.2~\text{m/s}^2$ means.
    \item If displacement is $12.0$ m, find average velocity.
  \end{enumerate}
  \vspace{0.5cm}

\item A drone flies north at $15.0~\text{m/s}$ for $36.0~\text{s}$.
  \begin{enumerate}[itemsep=1cm]
    \item Convert $15.0~\text{m/s}$ to km/h.
    \item Find distance traveled.
    \item If south is negative, what sign would you assign to this displacement?
  \end{enumerate}
  \vspace{0.5cm}

\end{enumerate}
